## Batch 2 Frontend: Forbidden Terms UI + Style Rules Page + Translation Editor Integration

Read CONTRIBUTING.md first.

### Context
Backend is already done:
- StyleRule + StyleOverride entities exist
- ForbiddenCheckController at /api/forbidden-check (POST: {text, languageCode, workspaceId} → matches)
- StyleCheckController at /api/style-check (POST: {text, projectId, contentType} → violations)
- StyleRulesController at /api/projects/{projectId}/style-rules (CRUD)
- ContentItemLanguageTask has RequiresReview bool field
- GlossaryTerm has IsForbidden + ForbiddenReplacement fields (already in glossary page)

### Task 1: Style Rules API Client

Create `frontend/app/api/styleRulesClient.ts`:
```typescript
export interface StyleRuleDto { id: string; projectId: string; name: string; ruleType: string; pattern: string; scope: string; message: string; isActive: boolean; createdUtc: string }
export interface StyleViolation { ruleId: string; ruleName: string; message: string; ruleType: string }
export interface StyleOverrideDto { id: string; contentItemLanguageTaskId: string; styleRuleId: string; overriddenByEmail: string; createdUtc: string }

export const styleRulesClient = {
  list: (projectId: string) => apiRequest<StyleRuleDto[]>(`/projects/${projectId}/style-rules`),
  create: (projectId: string, rule: Partial<StyleRuleDto>) => apiRequest<StyleRuleDto>(`/projects/${projectId}/style-rules`, { method: 'POST', body: JSON.stringify(rule) }),
  update: (projectId: string, ruleId: string, rule: Partial<StyleRuleDto>) => apiRequest<StyleRuleDto>(`/projects/${projectId}/style-rules/${ruleId}`, { method: 'PUT', body: JSON.stringify(rule) }),
  delete: (projectId: string, ruleId: string) => apiRequest<void>(`/projects/${projectId}/style-rules/${ruleId}`, { method: 'DELETE' }),
  check: (text: string, projectId: string, contentType: string) => apiRequest<StyleViolation[]>('/style-check', { method: 'POST', body: JSON.stringify({ text, projectId, contentType }) }),
}
```
Import apiRequest from ~/api/client.
